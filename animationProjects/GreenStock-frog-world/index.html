//Simple Tween

function flap() {

var character = new TimelineLite();

     tl.to(".wings", 0.1, {scaleY:0.8, rotation:-5, repeat:7, yoyo:true, transformOrigin: "50% 50%", ease:Bounce.easeInOut}, "master2+=10");

     return tl;
}

function blinky() {

var character = new TimelineLite();

TweenMax.to(".eye", 0.1, {scaleY:0, repeat:3, yoyo:true, transformOrigin: "50% 50%", ease:Power4.easeInOut});

TweenMax.to(".antenna", 0.1, {scaleY:0, rotation:5, repeat:3, yoyo:true, transformOrigin: "50% 50%", ease:Power4.easeInOut});

     return tl;
}

var master = new TimelineLite();
master.add(blinky());

master.add(flap());
